    /*

    // 儲存整理後的配方資料

    const emptyState = document.getElementById("emptyState");

    /*
      顯示狀態訊息
      type 可使用：success / error / info
    */
    function showStatus(element, message, type = "info") {
      element.textContent = message;
      element.className = `status show ${type}`;
    }

    function hideStatus(element) {
      element.textContent = "";
      element.className = "status";
    }

    /*
      文字正規化：
      - 去除前後空白
      - 轉小寫
      - 移除多餘空格
      讓搜尋更寬鬆
    */
    function normalizeText(text) {
      return String(text || "")
        .trim()
        .toLowerCase()
        .replace(/\s+/g, "");
    }

    /*
      簡易 CSV 解析器
      支援：
      - 逗號分隔
      - 雙引號包住含逗號的內容
      - 換行
    */
    function parseCSV(csvText) {
      const rows = [];
      let currentRow = [];
      let currentCell = "";
      let insideQuotes = false;

      for (let i = 0; i < csvText.length; i++) {
        const char = csvText[i];
        const nextChar = csvText[i + 1];

        if (char === '"' && insideQuotes && nextChar === '"') {
          currentCell += '"';
          i++;
        } else if (char === '"') {
          insideQuotes = !insideQuotes;
        } else if (char === "," && !insideQuotes) {
          currentRow.push(currentCell.trim());
          currentCell = "";
        } else if ((char === "\n" || char === "\r") && !insideQuotes) {
          if (char === "\r" && nextChar === "\n") i++;

          currentRow.push(currentCell.trim());

          if (currentRow.some(cell => cell !== "")) {
            rows.push(currentRow);
          }

          currentRow = [];
          currentCell = "";
        } else {
          currentCell += char;
        }
      }

      if (currentCell || currentRow.length > 0) {
        currentRow.push(currentCell.trim());
        if (currentRow.some(cell => cell !== "")) {
          rows.push(currentRow);
        }
      }

      return rows;
    }

    /*
      將 CSV rows 轉成物件陣列
    */
    function rowsToObjects(rows) {
      if (!rows || rows.length < 2) {
        throw new Error("CSV 內容不足，請至少提供標題列與一筆資料。");
      }

      const headers = rows[0].map(header => header.trim());
      const dataRows = rows.slice(1);

      return dataRows.map(row => {
        const obj = {};
        headers.forEach((header, index) => {
          obj[header] = row[index] || "";
        });
        return obj;
      });
    }

    /*
      找出可能的欄位名稱
      這裡做寬鬆比對，避免 CSV 欄位名稱稍微不同就讀不到
    */
    function findField(row, candidates) {
      const keys = Object.keys(row);

      for (const candidate of candidates) {
        const foundKey = keys.find(key => normalizeText(key) === normalizeText(candidate));
        if (foundKey) return row[foundKey];
      }

      for (const candidate of candidates) {
        const foundKey = keys.find(key => normalizeText(key).includes(normalizeText(candidate)));
        if (foundKey) return row[foundKey];
      }

      return "";
    }

    /*
      將原始 CSV 物件整理成配方結構
      支援兩種資料：
      1. 一列一個成分：相同名稱會被合併
      2. 一列一個配方：成分欄可能包含多個成分文字
    */
    function normalizeRecipes(rawRows) {
      const recipeMap = new Map();

      rawRows.forEach((row, index) => {
        const name = findField(row, ["名稱", "配方名稱", "藥膏名稱", "精油膏名稱", "name", "recipe"]);
        const purpose = findField(row, ["目的", "功效", "心願", "標籤", "分類", "用途", "purpose", "tag"]);
        const ingredient = findField(row, ["成分", "成份", "材料", "ingredient"]);
        const dose = findField(row, ["劑量", "份量", "用量", "dose", "amount"]);
        const note = findField(row, ["備註", "說明", "note", "description"]);

        if (!name && !ingredient && !purpose) return;

        const recipeName = name || `未命名配方 ${index + 1}`;

        if (!recipeMap.has(recipeName)) {
          recipeMap.set(recipeName, {
            name: recipeName,
            purpose: purpose || "",
            note: note || "",
            ingredients: []
          });
        }

        const recipe = recipeMap.get(recipeName);

        if (!recipe.purpose && purpose) recipe.purpose = purpose;
        if (!recipe.note && note) recipe.note = note;

        if (ingredient) {
          recipe.ingredients.push({
            name: ingredient,
            dose: dose || ""
          });
        }
      });

      return Array.from(recipeMap.values()).filter(recipe => {
        return recipe.name || recipe.purpose || recipe.ingredients.length > 0;
      });
    }

    /*
      讀取 CSV 檔案
    */
    csvFileInput.addEventListener("change", event => {
      const file = event.target.files[0];

      if (!file) return;

      if (!file.name.toLowerCase().endsWith(".csv")) {
        showStatus(fileStatus, "請上傳 CSV 檔案格式。", "error");
        return;
      }

      const reader = new FileReader();

      reader.onload = function(e) {
        try {
          const csvText = e.target.result;
          const rows = parseCSV(csvText);
          const objects = rowsToObjects(rows);
          recipes = normalizeRecipes(objects);

          if (recipes.length === 0) {
            showStatus(fileStatus, "沒有讀取到有效配方，請確認 CSV 是否包含名稱、目的或成分欄位。", "error");
            return;
          }

          showStatus(fileStatus, `已成功載入 ${recipes.length} 筆配方。`, "success");
          hideStatus(searchStatus);
          renderEmptyState();
        } catch (error) {
          showStatus(fileStatus, `讀取失敗：${error.message}`, "error");
        }
      };

      reader.onerror = function() {
        showStatus(fileStatus, "檔案讀取失敗，請重新上傳一次。", "error");
      };

      reader.readAsText(file, "UTF-8");
    });

    function handleSearch() {
      const keyword = searchInput.value.trim();

      resultContainer.innerHTML = "";
      recommendContainer.innerHTML = "";

      if (!keyword) {
        showStatus(searchStatus, "請先輸入想搜尋的目的或成分，例如：財富、檀香精油。", "error");
        emptyState.style.display = "block";
        return;
      }

      if (recipes.length === 0) {
        showStatus(searchStatus, "請先上傳 CSV，或點擊「載入範例資料」後再搜尋。", "error");
        emptyState.style.display = "block";
        return;
      }

      const normalizedKeyword = normalizeText(keyword);

      const matchedRecipes = recipes.filter(recipe => {
        const searchableText = [
          recipe.name,
          recipe.purpose,
          recipe.note,
          ...recipe.ingredients.map(item => `${item.name}${item.dose}`)
        ].join(" ");

        return normalizeText(searchableText).includes(normalizedKeyword);
      });

      if (matchedRecipes.length === 0) {
        showStatus(searchStatus, `找不到與「${keyword}」相關的配方，建議換個關鍵字，例如目的、精油名稱或植物油名稱。`, "info");

        `;
        return;
      }

      hideStatus(searchStatus);
      emptyState.style.display = "none";

      renderRecipes(matchedRecipes, resultContainer);

      const recommendations = getRecommendations(matchedRecipes);
      renderRecipes(recommendations, recommendContainer, true);

        recommendContainer.innerHTML = `
          <div class="empty-state">
            目前沒有找到可延伸製作的其他配方。<br>
            之後資料變多時，系統會依照相同成分自動推薦。
          </div>
        `;
      }
    }

    /*
      根據搜尋結果的成分，推薦其他配方
      邏輯：
      - 取出已命中配方中的所有成分
      - 找出其他也包含其中任一成分的配方
      - 依照共用成分數排序
    */
    function getRecommendations(matchedRecipes) {
      const matchedNames = new Set(matchedRecipes.map(recipe => recipe.name));

      const ownedIngredients = new Set();

      matchedRecipes.forEach(recipe => {
        recipe.ingredients.forEach(item => {
          ownedIngredients.add(normalizeText(item.name));
        });
      });

      const scored = recipes
        .filter(recipe => !matchedNames.has(recipe.name))
        .map(recipe => {
          const sharedIngredients = recipe.ingredients.filter(item => {
            return ownedIngredients.has(normalizeText(item.name));
          });

          return {
            recipe,
            score: sharedIngredients.length,
            sharedIngredients
          };
        })
        .filter(item => item.score > 0)
        .sort((a, b) => b.score - a.score);

      return scored.map(item => ({
        ...item.recipe,
        sharedIngredients: item.sharedIngredients
      }));
    }

    /*
      將配方卡片渲染到畫面
    */
    function renderRecipes(recipeList, container, isRecommendation = false) {
      container.innerHTML = "";

      recipeList.forEach(recipe => {
        const card = document.createElement("article");
        card.className = "card";

        const purposeTags = splitTags(recipe.purpose);

        const sharedHTML = isRecommendation && recipe.sharedIngredients
          ? `
            <div class="tag-row">
              ${recipe.sharedIngredients.map(item => `<span class="tag">已擁有：${escapeHTML(item.name)}</span>`).join("")}
            </div>
          `
          : "";

        card.innerHTML = `
          <h3>${escapeHTML(recipe.name)}</h3>

          ${recipe.purpose ? `<p class="purpose">${escapeHTML(recipe.purpose)}</p>` : ""}

          ${purposeTags.length > 0 ? `
            <div class="tag-row">
              ${purposeTags.map(tag => `<span class="tag">${escapeHTML(tag)}</span>`).join("")}
            </div>
          ` : ""}

          ${sharedHTML}

          ${recipe.note ? `<p class="purpose">${escapeHTML(recipe.note)}</p>` : ""}

          <ul class="ingredient-list">
            ${recipe.ingredients.map(item => `
              <li>
                <span class="ingredient-name">${escapeHTML(item.name)}</span>
                <span class="ingredient-dose">${escapeHTML(item.dose || "未填劑量")}</span>
              </li>
            `).join("")}
          </ul>
        `;

        container.appendChild(card);
      });
    }

    /*
      將目的欄位拆成標籤
    */
    function splitTags(text) {
      if (!text) return [];

      return text
        .split(/[,，、｜|/／\s]+/)
        .map(tag => tag.trim())
        .filter(Boolean)
        .slice(0, 8);
    }

    /*
      清除畫面
    */
    clearBtn.addEventListener("click", () => {
      searchInput.value = "";
      resultContainer.innerHTML = "";
      recommendContainer.innerHTML = "";
      hideStatus(searchStatus);
      renderEmptyState();
    });

    /*
      初始空狀態
    */
    function renderEmptyState() {
      resultContainer.innerHTML = "";
      recommendContainer.innerHTML = "";

      emptyState.style.display = "block";
      emptyState.innerHTML = `
        請輸入目的或成分後搜尋。<br>
        ${recipes.length > 0 ? `目前已載入 ${recipes.length} 筆配方。` : "請先上傳 CSV，或點擊「載入範例資料」體驗搜尋。"}
      `;
    }

    /*
      避免 CSV 內容中有特殊字元導致 HTML 注入
    */
    function escapeHTML(value) {
      return String(value || "")
        .replaceAll("&", "&amp;")
        .replaceAll("<", "&lt;")
        .replaceAll(">", "&gt;")
        .replaceAll('"', "&quot;")
        .replaceAll("'", "&#039;");
    }

    renderEmptyState();
  </script>
